# 1) WINSORISE (clip) arm-speed at 97.5 % → armSpeed_cap

def winsorise_top(series, q=0.975):
    upper = np.nanpercentile(series, q*100)
    return series.clip(upper=upper)

df_raw['armSpeed_cap'] = df_raw.groupby('pitcher')['armSpeed'].transform(winsorise_top)

latest_by = df_raw.groupby('pitcher')['date_corr'].max()

def recent_subset(grp, days=10):
    latest = latest_by.loc[grp.name]
    cutoff = latest - pd.Timedelta(days=days-1)
    return grp.loc[grp['date_corr'] >= cutoff]

def recent_pct(p):
    def _func(grp):
        rec = recent_subset(grp)
        if rec['armSpeed_cap'].notna().sum() >= 25:
            return np.nanpercentile(rec['armSpeed_cap'], p)
        return np.nanpercentile(grp['armSpeed_cap'], p)
    return _func

# ────────────────────────────────────────────────────────────────
# 3) Torque→Speed interpolation (per pitcher) —————————————
# ────────────────────────────────────────────────────────────────
RECENT_DAYS = 10
BIN, MIN_THROWS = 3, 10
torque_interp = {}

tmp = df_raw.dropna(subset=['armSpeed','torque']).copy()
tmp['t_bin'] = (tmp['torque'] / BIN).round() * BIN
summary = (
    tmp.groupby(['pitcher','t_bin'])
       .agg(n=('armSpeed','size'), spd=('armSpeed','mean'))
       .reset_index()
)
for p, g in summary.groupby('pitcher'):
    g2 = g[g['n'] >= MIN_THROWS].sort_values('t_bin')
    if len(g2) >= 2:
        torque_interp[p] = {'t': g2['t_bin'].values, 's': g2['spd'].values}

def interp_speed(pitcher, torque):
    d = torque_interp.get(pitcher)
    if d is None: return np.nan
    return float(np.interp(torque, d['t'], d['s'], left=d['s'][0], right=d['s'][-1]))

# ────────────────────────────────────────────────────────────────
# 4) Impute missing armSpeed → armSpeed_imp —————————————
#    • Game → 90th pct, Bullpen → 85th pct, then torque, then median
# ────────────────────────────────────────────────────────────────
def impute_speed(row):
    if not np.isnan(row['armSpeed']):
        return row['armSpeed']
    p   = row['pitcher']
    tag = row['tag'] if isinstance(row['tag'], str) else ''
    if 'Game' in tag:
        return pct90_by.get(p, np.nan)
    if 'Bullpen' in tag:
        return pct85_by.get(p, np.nan)

    # 1) try torque interpolation
    t = row.get('torque')
    if not np.isnan(t):
        val = interp_speed(p, t)
        if not np.isnan(val):
            return val

    # 2) fallback → median armSpeed over last RECENT_DAYS for this pitcher
    grp    = df_raw[df_raw['pitcher'] == p]
    latest = latest_by[p]
    cutoff = latest - pd.Timedelta(days=RECENT_DAYS-1)
    med    = grp.loc[grp['date_corr'] >= cutoff, 'armSpeed'].median()

    # 3) final fallback → global mean
    return float(med) if pd.notna(med) else df_raw['armSpeed'].mean()

df_raw['armSpeed_imp'] = df_raw.apply(impute_speed, axis=1)

def plot_absolute_velocity_decay(df_pulse, pitcher):
    subset = df_pulse[
        (df_pulse['converted_name'] == pitcher) &
        (df_pulse['COUNT'].notna()) & (df_pulse['COUNT'] != '')
    ]
    if subset.empty:
        print(f'No in-game data for {pitcher}')
        return

    fastball_pool = {'FA', 'SI', 'FC'}
    avail = fastball_pool & set(subset['PITCH_TYPE'].unique())
    if not avail:
        print(f'No fastball data for {pitcher}')
        return
    if 'FA' in avail:
        keep = {'FA', 'SI'} if 'SI' in avail else {'FA'}
    else:
        keep = {'SI'} if 'SI' in avail else {'FC'}

    fb = subset[subset['PITCH_TYPE'].isin(keep)]
    if fb.empty:
        print(f'No qualifying fastball data for {pitcher}')
        return

    # 1) Build historical & last-10 sessions
    curves = []
    for d, g in fb.groupby('date_corr'):
        valid = g[g['armSpeed'].notna()].sort_values('DATE_TIME')
        if not valid.empty:
            curves.append(valid['armSpeed'].reset_index(drop=True))
    if not curves:
        print(f'No qualifying fastball data for {pitcher}')
        return

    mat    = pd.concat(curves, axis=1).ffill().bfill()
    last_n = min(10, mat.shape[1])
    mat10  = mat.iloc[:, -last_n:]

    mean   = mat.mean(axis=1)
    q1, q3 = mat.quantile(0.25, axis=1), mat.quantile(0.75, axis=1)
    m10    = mat10.mean(axis=1)
    q1_10, q3_10 = mat10.quantile(0.25, axis=1), mat10.quantile(0.75, axis=1)

    # 2) Extract last-game session
    last_date = fb['date_corr'].max()
    g_last = fb[fb['date_corr'] == last_date].sort_values('DATE_TIME')
    if not g_last.empty:
        last_curve = g_last['armSpeed'].reset_index(drop=True)
        idx_last   = np.arange(1, len(last_curve) + 1)
    else:
        last_curve = None

    # 3) Plot
    idx = np.arange(1, len(mean) + 1)
    fig, ax = plt.subplots()

    # historical mean ± IQR
    ax.plot(idx, mean,  'k-', lw=2, label='Mean (hist)')
    ax.fill_between(idx, q1, q3, alpha=0.3, label='IQR (hist)')

    # last-10 mean ± IQR
    ax.plot(idx, m10,  'g-', lw=2, label='Mean (last10)')
    ax.fill_between(idx, q1_10, q3_10, alpha=0.3, color='green', label='IQR (last10)')

    # last-game raw curve
    if last_curve is not None:
        ax.plot(idx_last, last_curve, 'b--', lw=2, label=f'Last Game ({last_date})')

    ax.set_xlabel('Throw #')
    ax.set_ylabel('Velocity (mph)')
    ax.set_title(f'{pitcher} – In-Game Fastball Velocity Decay (n={mat.shape[1]})')
    ax.legend(loc='lower right')
    plt.tight_layout()
    plt.show()

def game_fastball_stats(df_pulse, pitcher, date):
    day = pd.Timestamp(date).date()
    sub = df_pulse[
        (df_pulse['converted_name'] == pitcher) &
        (df_pulse['date_corr'] == day) &
        (df_pulse['COUNT'].notna()) & (df_pulse['COUNT'] != '')
    ]
    if sub.empty:
        return (np.nan, np.nan)

    fb_pool = {'FA', 'SI', 'FC'}
    avail   = fb_pool & set(sub['PITCH_TYPE'].unique())
    if not avail:
        return (np.nan, np.nan)

    if 'FA' in avail:
        keep = {'FA', 'SI'} if 'SI' in avail else {'FA'}
    else:
        keep = {'SI'} if 'SI' in avail else {'FC'}

    fb = sub[sub['PITCH_TYPE'].isin(keep)]
    if fb.empty:
        return (np.nan, np.nan)

    return (fb['VELOCITY'].mean(), fb['VELOCITY'].max())

# Helper A – rolling z-score (28-day look-back, clipped ±3 SD)
# ────────────────────────────────────────────────────────────
ROLL_Z = 28
def _z(series):
    mu = series.rolling(ROLL_Z, min_periods=5).mean()
    sd = series.rolling(ROLL_Z, min_periods=5).std()
    return ((series - mu) / sd).clip(-3, 3)

#  Helper – pull in-game fastballs for a pitcher

def _fastball_subset(df, pitcher):
    """
    Return a Pulse-tracking slice with only *game* fastballs for `pitcher`,
    already filtered to FA (+SI), SI or FC exactly like the velocity-decay
    helpers do.  Empty DataFrame ⇒ nothing to plot.
    """
    fastball_pool = {'FA', 'SI', 'FC'}

    sub = df[
        (df['converted_name'] == pitcher) &
        df['PITCH_TYPE'].isin(fastball_pool) &
        df['COUNT'].notna() & (df['COUNT'] != '')
    ]

    avail = fastball_pool & set(sub['PITCH_TYPE'].unique())
    if not avail:
        return pd.DataFrame()          # no fastballs at all

    if 'FA' in avail:
        keep = {'FA', 'SI'} if 'SI' in avail else {'FA'}
    else:
        keep = {'SI'} if 'SI' in avail else {'FC'}

    out = sub[sub['PITCH_TYPE'].isin(keep)].copy()
    # sanity: make sure date_corr exists for grouping later
    if 'date_corr' not in out.columns:
        out['date_corr'] = out['DATE_TIME'].dt.date
    return out

def _scatter_plot(df, metric, metric_lbl, player_lbl):
    # 0) keep only the columns we need (no duplicates)
    if metric == "VELOCITY":
        cols = ["VELOCITY", "date_corr"]
    else:
        cols = [metric, "VELOCITY", "date_corr"]

    df = df[cols].dropna()
    if df.empty:
        return

    # 1) proper datetime & sort
    df["DATE"] = pd.to_datetime(df["date_corr"])
    df.sort_values("DATE", inplace=True)

    # 2) colour-coding by fastball velo
    cmap = cm.get_cmap("viridis")
    norm = Normalize(vmin=df["VELOCITY"].min(), vmax=df["VELOCITY"].max())

    # 3) 7-day rolling mean of the metric
    avg7 = (df.groupby("DATE")[metric]
              .mean()
              .rolling(7, min_periods=1)
              .mean())

    # 4) plot
    fig, ax = plt.subplots()

    sc = ax.scatter(df["DATE"], df[metric],
                    c=df["VELOCITY"], cmap=cmap, norm=norm,
                    s=15, alpha=.8, label="Throws")

    ax.plot(avg7.index, avg7.values, "k-", lw=2, label="7-day Avg")

    ax.set_title(f"{player_lbl} – {metric_lbl} In-Game")
    ax.set_xlabel("Date")
    ax.set_ylabel(metric_lbl)

    # ticks only on dates we actually have data for
    all_dates = pd.to_datetime(sorted(df["DATE"].unique()))
    ax.set_xticks(all_dates)
    ax.set_xticklabels([d.strftime("%b %d") for d in all_dates],
                       rotation=45, ha="right")

    ax.margins(x=0.01)

    # colour-bar
    cbar = fig.colorbar(sc, ax=ax, pad=.02)
    cbar.set_label("Fastball Velo (mph)")

    ax.legend()
    plt.tight_layout()
    plt.show()


def _forecast_transition(diff_series, window=5):
    """
    Estimate days until diff_series crosses zero by fitting a linear trend
    over the last `window` points.  Returns None if no crossing is predicted.
    """
    # grab the last `window` values as a 1-d numpy array:
    if isinstance(diff_series, pd.Series):
        y = diff_series.iloc[-window:].values
    else:
        # assume it’s already an ndarray
        y = diff_series[-window:]
    if len(y) < window:
        return None

    x = np.arange(len(y))
    slope = np.cov(x, y, bias=True)[0, 1] / np.var(x)
    current = y[-1]
    if slope < 0 and current > 0:
        return current / (-slope)
    if slope > 0 and current < 0:
        return current / (-slope)
    return None

def detect_velocity_cycles(df_raw, pitcher,
                           short_win=7, long_win=28,
                           raw_weight=0.6, imp_weight=0.4,
                           min_phase_len=5):
    """
    Combines raw & imputed arm-speed into a single signal,
    finds Load/Deload cycles, and enforces a minimum 5-day phase.
    """
    # 1) daily averages
    sub = df_raw[df_raw['pitcher'] == pitcher].dropna(subset=['armSpeed_imp','armSpeed'])
    daily_imp = sub.groupby('date_corr')['armSpeed_imp'].mean().sort_index()
    daily_raw = sub.groupby('date_corr')['armSpeed'].mean().sort_index()

    # 2) normalize
    dfn = pd.DataFrame({'imp':daily_imp, 'raw':daily_raw}).dropna()
    imp_z = (dfn['imp'] - dfn['imp'].mean()) / dfn['imp'].std()
    raw_z = (dfn['raw'] - dfn['raw'].mean()) / dfn['raw'].std()

    # 3) combined weighted signal
    signal = raw_weight * raw_z + imp_weight * imp_z

    # 4) rolling means & diff
    ma_short = signal.rolling(short_win, min_periods=1).mean()
    ma_long  = signal.rolling(long_win , min_periods=1).mean()
    diff     = ma_short - ma_long

    # 5) detect sign‐changes
    sign = np.sign(diff).replace(0, np.nan).ffill().bfill()
    transitions = diff.index[(sign.shift(1) != sign) & sign.shift(1).notna()]

    # 6) enforce min phase length
    filtered = [transitions[0]] if len(transitions) else []
    for t in transitions[1:]:
        if (t - filtered[-1]).days >= min_phase_len:
            filtered.append(t)

    # 7) compute filtered cycle lengths
    cycle_lengths = [
        (filtered[i] - filtered[i-1]).days
        for i in range(1, len(filtered))
    ]

    # 8) report
    median_cycle = float(np.median(cycle_lengths)) if cycle_lengths else np.nan
    current_trend = 'Load' if diff.iloc[-1] > 0 else 'Deload'
    forecast     = _forecast_transition(diff.values, window=short_win)

    return {
        'dates'        : dfn.index,
        'ma_short'     : ma_short,
        'ma_long'      : ma_long,
        'diff'         : diff,
        'transitions'  : filtered,
        'cycle_lengths': cycle_lengths,
        'median_cycle' : median_cycle,
        'current_trend': current_trend,
        'forecast_days': forecast
    }


def velocity_cycle_report(df_raw, pitcher, short_win=7, long_win=28):
    """
    Generate a velocity cycle report for a given pitcher.
    Displays a plot and prints a summary of the current trend,
    median cycle length and forecast.
    """
    info = detect_velocity_cycles(df_raw, pitcher, short_win, long_win)
    print(f"\n=== Velocity Cycle Report: {pitcher} ===")
    print(f"Current trend: {info['current_trend']} (diff = {info['diff'].iloc[-1]:.2f})")
    if info['forecast_days'] is not None:
        print(f"Estimated {info['forecast_days']:.1f} days until the next trend change.")
    else:
        print("No imminent trend change predicted based on the last few days.")
    if not np.isnan(info['median_cycle']):
        print(f"Median cycle length: {info['median_cycle']:.1f} days across {len(info['cycle_lengths'])} cycles.")
    else:
        print("Insufficient data to estimate cycle length.")
    # Plotting
    dates = pd.to_datetime(info['dates'])
    fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(12, 6), sharex=True)
    # Top panel: daily arm speed and moving averages
    ax1.plot(dates, info['daily'], '.', color='grey', alpha=0.5, label='Daily Avg Arm Speed')
    ax1.plot(dates, info['ma_short'], label=f'{short_win}-day MA', color='blue')
    ax1.plot(dates, info['ma_long'],  label=f'{long_win}-day MA', color='orange')
    ax1.set_ylabel('Arm Speed (deg/s)')
    ax1.set_title(f'{pitcher} – Arm Speed and Moving Averages')
    ax1.legend(loc='upper left')
    # Bottom panel: difference and phase shading
    diff = info['diff']
    diff_dates = pd.to_datetime(diff.index)
    ax2.plot(diff_dates, diff, color='purple', label='MA Difference')

    # Mark transition dates
    for d in info['transitions']:
        ax2.axvline(pd.to_datetime(d), linestyle='--', color='grey', linewidth=1)
    ax2.set_ylabel('MA diff (deg/s)')
    ax2.set_xlabel('Date')
    ax2.set_title('Load vs. Deload Phases (green = load, red = deload)')
    #  ── ADD GAME‐DAY TICKS ──────────────────────────
    # find every date where next_game>0 (i.e. today was a game)
    game_days = df_daily.index[df_daily['is_game']]

    # draw a little red “|” tick at the bottom of the diff‐panel
    ymin, _ = ax2.get_ylim()

        ax2.plot([d, d], [ymin, ymin + 0.02*(ax2.get_ylim()[1]-ymin)],

        raw_weight=0.6,
        imp_weight=0.4,
        zero_tol=0.02,
        min_phase_len=3,
        smooth_gap=False,
        gap_smooth_win=7,

        slope_win=None         # ← new: let me override which window to use for slope
    ):

    """
    Uses both raw and imputed Load-Index signals to detect Load/Deload,
    applies a zero‐tolerance, and skips any phase < min_phase_len days.
    """
    # 1) rolling means on raw & imp signals
    imp = df_daily['load_idx']
    raw = ((df_daily['spd_metric'] - df_daily['spd_metric'].mean())
           / df_daily['spd_metric'].std()).clip(-3, 3)

    ma_imp_s = imp.rolling(short_win, min_periods=1).mean()
    ma_imp_l = imp.rolling(long_win,  min_periods=1).mean()
    ma_raw_s = raw.rolling(short_win, min_periods=1).mean()
    ma_raw_l = raw.rolling(long_win,  min_periods=1).mean()

    # 2) combined diff
    diff_imp = ma_imp_s - ma_imp_l
    diff_raw = ma_raw_s - ma_raw_l
    diff     = raw_weight * diff_raw + imp_weight * diff_imp

    # optional smooth of gaps
    if smooth_gap:
        diff = diff.rolling(gap_smooth_win, 1).mean()

    # 3) zero‐tolerance
    diff[np.abs(diff) < zero_tol] = 0
    sign = np.sign(diff).replace(0, np.nan).ffill().bfill()

    # 4) find transitions
    transitions = diff.index[(sign.shift(1) != sign) & sign.shift(1).notna()]

    # 5) enforce min phase length
    filtered = [transitions[0]] if len(transitions) else []
    for t in transitions[1:]:
        if (t - filtered[-1]).days >= min_phase_len:
            filtered.append(t)

    # 6) separate phase lengths
    load_len, deload_len = [], []
    for i in range(1, len(filtered)):
        days = (filtered[i] - filtered[i-1]).days
        if sign.loc[filtered[i-1]] > 0:
            load_len.append(days)
        else:
            deload_len.append(days)

    # 7) medians
    median_load   = float(np.median(load_len))   if load_len   else np.nan
    median_deload = float(np.median(deload_len)) if deload_len else np.nan

    # grab the most recent diff value
    last_val = diff.iloc[-1]

    # 8) compute slope over the last S points
    W = slope_win or short_win
    y = diff.values[-W:]
    x = np.arange(len(y))
    slope = np.cov(x, y, bias=True)[0,1] / np.var(x)

    # 9) forecast days until diff hits zero if slope points toward zero
    if slope * last_val < 0:
        forecast = abs(last_val / slope)
    else:
        forecast = None

    # 10) enforce your “if we’re sitting smack at zero, call it now”
    if forecast is None and abs(last_val) < zero_tol:
        forecast = 0.0

    # define current phase
    current_trend = 'Load' if last_val > 0 else 'Deload'

    # compute slope over the last W points
    W = slope_win or short_win
    y = diff.values[-W:]
    x = np.arange(len(y))
    slope = np.cov(x, y, bias=True)[0,1] / np.var(x)

    # forecast days until diff hits zero
    if slope * last_val < 0:
        forecast = abs(last_val / slope)
    else:
        forecast = None

    # if we’re effectively at zero right now
    if forecast is None and abs(last_val) < zero_tol:
        forecast = 0.0

    # compute days into current phase
    if filtered:
        current_cycle_days = (diff.index[-1] - filtered[-1]).days + 1
    else:
        current_cycle_days = len(diff)

    return {
        'diff': diff,
        'transitions': filtered,
        'median_load': median_load,
        'median_deload': median_deload,
        'current_trend': current_trend,
        'forecast_days': forecast,
        'current_cycle_days': current_cycle_days
    }


def load_cycle_report(df_daily, player,
                      short_win=7, long_win=28,
                      **kwargs):
    info = detect_load_cycles(df_daily,
                               short_win=short_win,
                               long_win=long_win,
                               **kwargs)
    diff = info['diff']
    dates = pd.to_datetime(diff.index)

    # align df_daily so we can look up throws for every diff point
    aln = df_daily.reindex(dates)

    fig, ax = plt.subplots(figsize=(12, 4), sharex=True)

    # ── diff curve ────────────────────────────────────────────────────
    ax.plot(dates, diff, color='purple', lw=1.5, label='MA Difference')
    ax.axhline(0, color='black', lw=1)

    # ── build phase edges: [start, …transitions…, end] ────────────────
    edges = [dates[0]] + info['transitions'] + [dates[-1] + pd.Timedelta(days=1)]

    for start, end in zip(edges[:-1], edges[1:]):
        seg_mask = (dates >= start) & (dates < end)
        if not seg_mask.any():
            continue

        diff_seg = diff[seg_mask]

        # % GAME workload in this phase
        g_throws   = aln.loc[seg_mask, 'next_game'].sum()      # game throws
        total_thr  = aln.loc[seg_mask, 'total_throws'].sum()
        game_ratio = (g_throws / total_thr) if total_thr else 0.0
        game_ratio = np.clip(game_ratio, 0, 1)

        # choose colours
        is_load  = diff_seg.mean() >= 0
        base_c , dark_c = (('#b7e6b7', '#006400') if is_load
                           else ('#fcbaba', '#8b0000'))

        # lighter shade – whole area
        ax.fill_between(diff_seg.index, 0, diff_seg,
                        color=base_c, alpha=.5, lw=0)

        # darker shade – % GAME share
        ax.fill_between(diff_seg.index, 0, diff_seg * game_ratio,
                        color=dark_c, alpha=.8, lw=0)

    # mark phase transitions
    for t in info['transitions']:
        ax.axvline(pd.to_datetime(t), ls='--', color='grey', lw=1)

    # optional FB-velo overlay (kept from old version)
    if 'velo_7d' in aln.columns:
        ax2 = ax.twinx()
        ax2.plot(dates, aln['velo_7d'], color='dodgerblue', lw=1.5,
                 label='7-day FB velo')
        ax2.set_ylabel('FB Velo (mph)', color='dodgerblue')
        ax2.tick_params(axis='y', colors='dodgerblue')
        ax2.legend(loc='upper right')

    # tiny red tick for every game day
    ymin, _ = ax.get_ylim()
    for d in df_daily.index[df_daily['is_game']]:
        ax.plot([d, d], [ymin, ymin + 0.02 * (ax.get_ylim()[1] - ymin)],
                color='red', lw=1)

    ax.set_ylabel('MA diff')
    ax.set_xlabel('Date')
    ax.set_title('Load (green) vs Deload (red) — darker shade = % GAME workload')
    ax.legend(loc='upper left')
    ax.xaxis.set_major_locator(LOCATOR)
    ax.xaxis.set_major_formatter(FORMATTER)
    plt.setp(ax.get_xticklabels(), rotation=90, ha='center', fontsize=8)
    plt.tight_layout()
    plt.show()
# ========================================================================

    # console summary
    days_in = info['current_cycle_days']
    print(f"\n=== Load-Cycle Report: {player} ===")
    print(f"Current trend: {info['current_trend']} "
          f"(diff = {info['diff'].iloc[-1]:.2f}) – "
          f"{days_in} day{'s' if days_in!=1 else ''} into this phase")

    fd = info['forecast_days']
    if fd is not None:
        if fd == 0:
            print("Trend change happening now based on recent direction.")
        else:
            print(f"Estimated {fd:.1f} days until next change.")
    else:
        print("No imminent trend change predicted.")

    if not np.isnan(info['median_load']):
        print(f"Median Load   phase: {info['median_load']:.1f} d")
    if not np.isnan(info['median_deload']):
        print(f"Median Deload phase: {info['median_deload']:.1f} d")

    daily['vol_score']   = (daily['total_throws'] / vol90).clip(0, 1)

# Helper – outing-level history for READINESS
# ─────────────────────────────────────────────────────────────
def _build_outing_rows(pitcher):
    daily_p    = daily_dict[pitcher]
    game_dates = daily_p.index[daily_p['is_game']]

    if game_dates.empty:
        cols = [
            'date',
            'avg_velo', 'max_velo',
            'days_rest', 'acwr_last',
            'last_day_label', 'last_avg_spd',
            'pattern_gap', 'tot_thr_between',
            'hi_ratio_between', 'sum_var_between',
            'avg_sp_between', 'p90_sp_between',
            'success'
        ]
        return pd.DataFrame(columns=cols)

    rows = []
    for d in game_dates:
        avg_v, max_v = game_fastball_stats(df_pulse, pitcher, d)
        pos = daily_p.index.get_loc(pd.to_datetime(d))

        # 1) days_rest
        if pos == 0:
            rest = 0
        else:
            raw  = daily_p['days_since_game'].iat[pos - 1]
            rest = 0 if pd.isna(raw) else int(raw)

        # 2) acwr_last, last_day_label, last_avg_spd
        if pos == 0:
            acwr_last = np.nan
            last_lbl  = 'NA'
            last_spd  = np.nan
        else:
            prev       = daily_p.iloc[pos - 1]
            acwr_last  = prev['acwr']
            last_lbl   = prev['label']
            last_spd   = prev['spd_metric']

        # 3) pattern_gap
        lbl_str   = daily_p['label'].astype(str).replace('nan', '')
        start_ix  = max(0, pos - rest)
        seq       = lbl_str.iloc[start_ix:pos].tolist()
        seq       = [s.replace('Medium','MED').replace('Low','LOW') for s in seq if s]
        pattern   = '-'.join(seq) if seq else 'NA'

        # 4) between-segment stats
        seg      = daily_p['segment'].iat[pos]
        prev_seg = between_dict[pitcher].query("segment == @seg - 1")
        if prev_seg.empty:
            tot_thr = np.nan
            hi_rat  = np.nan
            sum_var = np.nan
            avg_sp  = np.nan
            p90_sp  = np.nan
        else:
            b        = prev_seg.squeeze()
            tot_thr  = b['total_throws_β']
            hi_rat   = b['hi_ratio_β']
            sum_var  = b['sum_variation']
            avg_sp   = b['avg_speed']
            p90_sp   = b['p90_speed']

        rows.append({
            'date'             : d,
            'avg_velo'         : avg_v,
            'max_velo'         : max_v,
            'days_rest'        : rest,
            'acwr_last'        : acwr_last,
            'last_day_label'   : last_lbl,
            'last_avg_spd'     : last_spd,
            'pattern_gap'      : pattern,
            'tot_thr_between'  : tot_thr,
            'hi_ratio_between' : hi_rat,
            'sum_var_between'  : sum_var,
            'avg_sp_between'   : avg_sp,
            'p90_sp_between'   : p90_sp
        })

    out = pd.DataFrame(rows)

    # ⮕ force those four columns to float (no more “object” dtype)
    for col in ['days_rest','acwr_last','tot_thr_between','hi_ratio_between']:
        out[col] = pd.to_numeric(out[col], errors='coerce')

    # Compute “success” as before
    if not out['avg_velo'].dropna().empty:
        thr = np.nanpercentile(out['avg_velo'], 80)
        out['success'] = (out['avg_velo'] >= thr).astype(int)
    else:
        out['success'] = 0

    return out

def _velo_arrow(v7, v30):
    if pd.isna(v7) or pd.isna(v30):

        return ' ▲'
    if v7 < v30:

    return ' •'

def _ng_game_bar(val, red='#d73027'):
    """
    Cell contains something like '78% / 22%'   (NG% / Game%)

    """

# ─────────────────────────────────────────────────────────────
# Helper 4 – mean fastball velo (same rules as decay plots)
#           • COUNT *not* empty (game only)
#           • keep  FA (+ SI) or SI or FC
# ─────────────────────────────────────────────────────────────
_FASTBALL_TYPES = {'FA', 'SI', 'FC'}

def _mean_fb_vel(df, name, start_date):
    sub = df.loc[
        (df['converted_name'] == name) &
        (df['date_corr']      >= start_date) &
        (df['COUNT'].notna()) & (df['COUNT'] != '')
    ]
    if sub.empty:
        return np.nan

    avail = _FASTBALL_TYPES & set(sub['PITCH_TYPE'].unique())
    if not avail:
        return np.nan

    if 'FA' in avail:
        keep = {'FA', 'SI'} if 'SI' in avail else {'FA'}
    else:
        keep = {'SI'} if 'SI' in avail else {'FC'}

    fb = sub[sub['PITCH_TYPE'].isin(keep)]
    return fb['VELOCITY'].mean() if not fb.empty else np.nan

def _off_days(name, start_date, end_date):
    """How many days in [start_date .. end_date] have zero throws?"""
    # all calendar days in window
    alldays = pd.date_range(start_date, end_date, freq='D').date

    # any throw recorded in df_raw marks the day as 'on‑throwing'
    mask = (df_raw['pitcher'] == name) & \
           (df_raw['date_corr'].between(start_date, end_date))
    ondays = set(df_raw.loc[mask, 'date_corr'])

    return len(alldays) - len(ondays)


def _bullpen_throws(name, start_date):
    """Total # of bullpen throws in df_pulse since start_date."""
    is_bp = df_pulse['COUNT'].isna() | \
            (df_pulse['COUNT'].astype(str).str.strip() == '')
    mask = (df_pulse['converted_name'] == name) & \
           (df_pulse['date_corr'] >= start_date) & is_bp
    return df_pulse.loc[mask].shape[0]

# Tank v2  – includes ACWR, rest-quota, composite bonus/penalty
# ────────────────────────────────────────────────────────────────
SEASON_CUTOFF = (4, 1)          # 1 Apr

def _calibrate_rest_acwr(pitcher):
    """
    Returns
    -------
    opt_rest : int        – median rest-days (floor = 2) when velo ≥ 90th-pct
    acwr_lo  : float      – minimum ACWR that still produced velo ≥ 80th-pct
    acwr_hi  : float      – maximum ACWR that still produced velo ≥ 80th-pct
    """
    outings = _build_outing_rows(pitcher)
    if outings.empty:
        # sensible, conservative fall-backs
        return 5, 0.80, 1.30

    # ── keep only in-season games (after 1 Apr) ───────────────────────
    def _after_cutoff(ts):
        return (ts.month, ts.day) >= SEASON_CUTOFF

    data = outings[outings['date'].apply(_after_cutoff)]
    if data.empty:                       # all spring games? – then use all
        data = outings

    # ── 1) Optimal REST (median of velo ≥ 90th-pct & rest ≥ 2 d) ──────
    v90 = data['avg_velo'].quantile(.90)
    rest_pool = data.loc[(data['avg_velo'] >= v90) & (data['days_rest'] >= 2),
                         'days_rest']
    opt_rest = rest_pool.median()
    if np.isnan(opt_rest):               # too few outings
        opt_rest = data['days_rest'].median()
    opt_rest = max(int(round(opt_rest)), 2)   # hard floor = 2

    # ── 2) ACWR **range** for outings with velo ≥ 80th-pct ────────────
    v80        = data['avg_velo'].quantile(.80)
    acwr_pool  = data.loc[data['avg_velo'] >= v80, 'acwr_last'].dropna()

    if acwr_pool.empty:
        # fall back to overall median ±0.05, or classic 0.8–1.3 window
        mid = data['acwr_last'].median()
        if np.isnan(mid):
            acwr_lo, acwr_hi = 0.80, 1.30
        else:
            acwr_lo = max(mid - 0.05, 0)   # keep it positive
            acwr_hi = mid + 0.05
    else:
        acwr_lo = acwr_pool.min()
        acwr_hi = acwr_pool.max()

    return opt_rest, float(acwr_lo), float(acwr_hi)


def simulate_tank_v2(daily_df, pitcher,
                     acwr_weight=10,
                     comp_weight =  6,
                     base_floor=10):

    # ── 1) pitcher-specific parameters ─────────────────────────
    # 1) get rest-day target and ACWR range
    opt_rest, acwr_lo, acwr_hi = _calibrate_rest_acwr(pitcher)

    # 2) pick a single “target ACWR” to steer the tank toward
    opt_acwr = 0.5 * (acwr_lo + acwr_hi)     # midpoint of the range
    base = 100.0 / opt_rest                    # pts added per full rest day
    tank, no90 = 100.0, 0
    ready = []

    # ── 2) iterate one calendar day at a time ─────────────────
    for _, row in daily_df.iterrows():

        # ── 2 A) GAME DAY  ⇒ empty the tank ───────────────────
        if row.is_game:
            tank = 0.0                         # ← THE ONLY REAL CHANGE
            no90 = 0

        # ── NON-GAME DAY ───────────────────────────────────────────
        else:
            # 1) unconditional recharge proportional to how light the day was
            rest_factor = 1 - row['vol_score']          # 1.0 → 0 throws, 0 → vol90
            tank += base * rest_factor

            # 2) small composite nudge (max ±½ base)
            delta = (0.50 - row['composite'])           #  + when easy, − when heavy
            tank += comp_weight * delta                 # range ≈ ±10

            # 3) ACWR correction (max ±½ base)
            if not np.isnan(row.acwr):
                tank += acwr_weight * (opt_acwr - row.acwr)

            # 4) “no-90th” fatigue after 3 straight days
            got90 = (row.hi_ratio >= 0.10)
            no90  = 0 if got90 else no90 + 1
            if no90 >= 3:
                tank -= base * 0.50                      # half a rest-day’s worth

        tank = np.clip(tank, base_floor, 100)
        ready.append(tank)

    daily_df['tank_readiness_v2'] = ready
    return daily_df

                                  if g['spd_metric'].notna().sum() else np.nan,
                     'sum_variation': np.nan,  # placeholder
                     'total_throws_β': g['total_throws'].sum(),
                     'hi_ratio_β': g['raw_high_intent'].sum() / max(g['total_throws'].sum(), 1)
                 }))
                 .reset_index(drop=True)
    )
    between_dict[p] = btmp

    # count high_intent days for formula
    hi_mask   = raw_all['armSpeed_imp'] >= pct75_by[p]
    hi_counts = (
        raw_all.loc[hi_mask]
              .groupby('date_corr')
              .size()
              .rename('high_intent')
    )
    daily_tmp['high_intent'] = daily_tmp.index.map(hi_counts).fillna(0)

    daily_dict[p] = daily_tmp

# ──────────────────────────────────────────────────────────────────
# 4) Fill out daily_dict[p] with “Tank readiness” only
# ──────────────────────────────────────────────────────────────────
for p in pitchers:
    daily_p = daily_dict[p].drop(columns="date").reset_index()
    # hist_p  = _build_outing_rows(p)  # not used here except inside calibrate

    # Force numeric on the four features (needed by calibrate)
    hist_p = _build_outing_rows(p)
    for col in ['days_rest','acwr_last','tot_thr_between','hi_ratio_between']:
        hist_p[col] = pd.to_numeric(hist_p[col], errors='coerce')

    # ─── BUILD ALL DAILY-SCORES ───────────────────────────────────
    df_dp = daily_p.copy()

    # 1) day-to-day workload variation
    df_dp['throw_var'] = df_dp['total_throws'].diff().abs().fillna(0)
    between_dict[p]['sum_variation'] = df_dp.groupby('segment')['throw_var'].sum().values

    # 2) hi_ratio = high_intent / total_throws
    df_dp['hi_ratio'] = df_dp['high_intent'] / df_dp['total_throws'].replace(0, np.nan)

    # 3) daily_intent_weight from df_raw['intent_weight']
    intent_per_day = df_raw.groupby('date_corr')['intent_weight'].sum()
    df_dp['daily_intent_weight'] = df_dp['date'].map(intent_per_day).fillna(0)

    # 4) hybrid_intent = daily_intent_weight * hi_ratio
    df_dp['hybrid_intent'] = df_dp['daily_intent_weight'] * df_dp['hi_ratio']

    # 5) 28-day rolling 10th & 90th for hybrid_intent
    ROLL   = 28
    hyb_p10 = df_dp['hybrid_intent'].rolling(ROLL, min_periods=5).quantile(0.10)
    hyb_p90 = df_dp['hybrid_intent'].rolling(ROLL, min_periods=5).quantile(0.90)
    den_h   = (hyb_p90 - hyb_p10).replace(0, np.nan)

    # 6) intent_score = ((hybrid_intent - p10) / (p90 - p10)).clip(0,1)
    df_dp['intent_score'] = ((df_dp['hybrid_intent'] - hyb_p10) / den_h) \
                              .clip(0,1).fillna(0)

    # 7) speed_score = 0.25 + 0.75 * normalized(spd_metric)
    spd_mask = df_dp['spd_metric'].notna()
    spd_min  = df_dp.loc[spd_mask, 'spd_metric'].min()
    spd90    = df_dp.loc[spd_mask, 'spd_metric'].quantile(0.90)
    spd_den  = max(spd90 - spd_min, 1)

    spd_raw              = ((df_dp['spd_metric'] - spd_min) / spd_den).clip(0, 1)
    df_dp['speed_score'] = 0.25 + 0.75 * spd_raw
    df_dp['speed_score'] = df_dp['speed_score'].fillna(0.25)

    # 8) game_bonus = 0.10 on days where next_game > 0
    df_dp['game_bonus'] = (df_dp['next_game'] > 0).astype(float) * 0.10

    # 9) vol_score = (total_throws / 90th-percentile_of_total_throws).clip(0,1)
    vol_mask = df_dp['total_throws'] > 0
    vol90    = df_dp.loc[vol_mask, 'total_throws'].quantile(0.90)
    vol90    = vol90 if vol90 > 0 else 1
    df_dp['vol_score'] = (df_dp['total_throws'] / vol90).clip(0, 1).fillna(0)

    # ← INSERT “composite” here
    df_dp['composite'] = (
          0.50 * df_dp['vol_score']
        + 0.35 * df_dp['intent_score']
        + 0.25 * df_dp['speed_score']
        +        df_dp['game_bonus']
    ).clip(0, 1)

    # ── NOW SIMULATE TANK READINESS ──────────────────────────────────
    df_dp = simulate_tank_v2(df_dp, p)

    # save back into daily_dict[p]
    daily_dict[p] = df_dp.set_index('date')


    # ---------------------------------------------------------------
    # 10) ACWR (7-day acute / 28-day chronic)  ← NEW, must come first
    # ---------------------------------------------------------------
    df_dp['acute']   = df_dp['composite'].ewm(span=7 , adjust=False).mean()
    df_dp['chronic'] = df_dp['composite'].ewm(span=28, adjust=False).mean()
    df_dp['acwr']    = df_dp['acute'] / df_dp['chronic']

    # write the ACWR-populated frame back to daily_dict
    daily_dict[p] = df_dp.set_index('date')

    # now it’s safe to calibrate / simulate – ACWR data exist!
    df_dp = simulate_tank_v2(df_dp, p)

    # finally, store the version that now also has tank readiness
    daily_dict[p] = df_dp.set_index('date')

def _prep_xy(df, target):
    df = df.dropna(subset=[target])
    if df.empty:
        return pd.DataFrame(), pd.Series(dtype=float)

    X = df.drop(columns=['avg_velo', 'max_velo', 'date'])
    X = pd.get_dummies(
            X,
            columns=['last_day_label', 'pattern_gap'],
            dummy_na=True
        ).apply(pd.to_numeric, errors='coerce').fillna(0)

    y = df[target]
    return X, y

def _train_rf(X,y):
    grid={'n_estimators':[100,200],'max_depth':[8,16]}
    gs=GridSearchCV(RandomForestRegressor(random_state=42),
                    grid,cv=3,n_jobs=-1)
    gs.fit(X,y)
    return gs.best_estimator_

def _train_xgb(X,y):
    grid={'n_estimators':[200,400],'max_depth':[5,8],'learning_rate':[0.05,0.1]}
    gs=GridSearchCV(XGBRegressor(random_state=42,verbosity=0),
                    grid,cv=3,n_jobs=-1)
    gs.fit(X,y)
    return gs.best_estimator_

from itertools import product

feature_cols = None

def _align(mat, cols):
    return mat.reindex(columns=cols, fill_value=0)

def optimal_pattern(pitcher, gap, rf_model, xgb_model, X_template, cols):
    labels   = ['LOW', 'MED', 'HIGH']
    best_pat, best_pred = None, -np.inf

    for combo in product(labels, repeat=gap):
        pat = "-".join(combo)
        X = X_template.copy()

        pat_cols = [c for c in X.columns if c.startswith("pattern_gap_")]
        X[pat_cols] = 0
        col = f"pattern_gap_{pat}"
        if col not in X.columns:
            X[col] = 0
        X[col] = 1
        X = _align(X, cols)

        rf_pred  = rf_model.predict(X)[0]
        xgb_pred = xgb_model.predict(X)[0]
        ens_pred = (rf_pred + xgb_pred) / 2

        score = ens_pred + 0.01 * len(set(combo))

        if score > best_pred:
            best_pat, best_pred = pat, score

        # second pass to re-zero out all other pattern_gap_ columns:
        pat_cols = [c for c in X.columns if c.startswith('pattern_gap_')]
        X[pat_cols] = 0
        col_name = f'pattern_gap_{pat}'
        if col_name not in X.columns:
            X[col_name] = 0
        X[col_name] = 1
        X = _align(X, cols)
        y_hat = (rf_model.predict(X)[0] + xgb_model.predict(X)[0]) / 2
        if y_hat > best_pred:
            best_pat, best_pred = pat, y_hat

    return best_pat, best_pred

def _importance(model, feat_names, top=None):
    imp = model.feature_importances_
    idx = np.argsort(imp)[::-1]
    if top is not None:
        idx = idx[:top]
    return [(feat_names[i], imp[i]) for i in idx]

def _propose_schedule(daily_p, start_date, pattern):
    if (pattern == "NA") or (not pattern):
        return pd.DataFrame()

    outings = _build_outing_rows(pitcher_fc_dd.value)
    if outings.empty:
        top = pd.DataFrame()
    else:
        thresh = outings["max_velo"].quantile(0.75)
        top    = outings[outings["max_velo"] >= thresh]

    rest_days      = daily_p[daily_p['label'].isin(['Low', 'Medium', 'High'])]
    throws_med     = rest_days.loc[rest_days['label'] == 'Medium', 'total_throws'].median()
    hi_ratio_med   = (
        rest_days.loc[rest_days['label'] == 'Medium', 'high_intent'].sum()
        / max(rest_days.loc[rest_days['label'] == 'Medium', 'total_throws'].sum(), 1)
    )

    mult   = {"LOW": 0.4, "MED": 0.7, "HIGH": 1.0}
    labels = pattern.split("-")
    days   = pd.date_range(
                start_date - pd.Timedelta(days=len(labels)),
                periods=len(labels),
                freq="D"
            )

    rows = []
    for d, lbl in zip(days, labels):
        tot = int(throws_med * mult[lbl])
        hi  = int(tot * hi_ratio_med)
        rows.append({
            "Date"        : d.date(),
            "Day Label"   : lbl,
            "Total Throws": tot,
            "High Intent" : hi
        })

    return pd.DataFrame(rows)

import time, sys

def _status(msg):
    print(msg.ljust(50), end='\r')
    sys.stdout.flush()

def run_forecast(_):
    global feature_cols
    feature_cols = None
    with out_fc:
        out_fc.clear_output()
        t0 = time.time()
        p      = pitcher_fc_dd.value
        start_d = date_picker.value
        if (p is None) or (start_d is None):
            print("Select a pitcher and a date.")
            return

        _status("⏳  training models ...")
        hist = _build_outing_rows(p)
        if hist.empty:
            print("Not enough completed outings for modelling.")
            return

        models, preds = {}, {}
        for tgt in ["avg_velo", "max_velo"]:
            X, y = _prep_xy(hist, tgt)
            if feature_cols is None:
                feature_cols = X.columns.tolist()
            if y.empty:
                print(f"Not enough non-NaN {tgt} targets.")
                return
            X_last = X.iloc[[-1]].copy()
            preds[tgt] = {"X_last": X_last}

            rf  = _train_rf(X, y)
            xgb = _train_xgb(X, y)
            models[tgt] = {"rf": rf, "xgb": xgb}

        _status("⏳  searching optimal pattern ...")
        last_d   = df_raw.loc[df_raw["pitcher"] == p, "date_corr"].max()
        rest_gap = max((start_d - last_d).days - 1, 1)

        for tgt in preds:
            X_tmpl = preds[tgt]["X_last"].copy()
            X_tmpl = _align(X_tmpl, feature_cols)
            X_tmpl["days_rest"] = rest_gap
            preds[tgt]["X_tmpl"] = X_tmpl

        best_pat_avg, best_ens_avg = optimal_pattern(
            p,
            rest_gap,
            models["avg_velo"]["rf"],
            models["avg_velo"]["xgb"],
            preds["avg_velo"]["X_tmpl"].copy(),
            feature_cols
        )

        best_pat_max, best_ens_max = optimal_pattern(
            p,
            rest_gap,
            models["max_velo"]["rf"],
            models["max_velo"]["xgb"],
            preds["max_velo"]["X_tmpl"].copy(),
            feature_cols
        )

        print(f"🧮 Optimal {rest_gap}-day pattern for AVG_VELO: {best_pat_avg}")
        print(f"🧮 Optimal {rest_gap}-day pattern for MAX_VELO: {best_pat_max}\n")

        X_avg = preds["avg_velo"]["X_tmpl"].copy()
        for col in X_avg.columns:
            if col.startswith("pattern_gap_"):
                X_avg[col] = 0
        X_avg[f"pattern_gap_{best_pat_avg}"] = 1
        preds["avg_velo"]["X_now"] = _align(X_avg, feature_cols)

        X_max = preds["max_velo"]["X_tmpl"].copy()
        for col in X_max.columns:
            if col.startswith("pattern_gap_"):
                X_max[col] = 0
        X_max[f"pattern_gap_{best_pat_max}"] = 1
        preds["max_velo"]["X_now"] = _align(X_max, feature_cols)

        for tgt in ["avg_velo", "max_velo"]:
            rf  = models[tgt]["rf"]
            xgb = models[tgt]["xgb"]
            Xn  = preds[tgt]["X_now"]

            rf_pred  = rf.predict(Xn)[0]
            xgb_pred = xgb.predict(Xn)[0]
            ens_pred = (rf_pred + xgb_pred) / 2

            preds[tgt]["rf"]  = rf_pred
            preds[tgt]["xgb"] = xgb_pred
            preds[tgt]["ens"] = ens_pred

        daily_p = daily_dict[p]
        if rest_gap > 1:
            sched = _propose_schedule(daily_p, start_d, best_pat_avg)
            if "Date" in sched.columns:
                sched = sched[sched["Date"] < start_d]
        else:
            sched = pd.DataFrame()
        if not sched.empty:
            print("📋 Suggested Throw-Schedule:")
            display(sched.style.hide(axis="index"))
        elif rest_gap > 1:
            print("⚠️ No schedule suggestion (insufficient data).\n")

        for tgt in ["avg_velo", "max_velo"]:
            print(f"\n{tgt.upper()} – predictions (mph)")
            print(f"   RF   : {preds[tgt]['rf'] :.2f}")
            print(f"   XGB  : {preds[tgt]['xgb']:.2f}")
            print(f"   MEAN : {preds[tgt]['ens']:.2f}")
            _status("✓  prediction ready                             ")
            print(f"\n⌚ elapsed: {time.time()-t0:,.1f} s\n")

            for lab, model in [("RF", models[tgt]["rf"]), ("XGB", models[tgt]["xgb"])]:
                top_feats = _importance(model, preds[tgt]["X_now"].columns, top=10)
                print(f"   · {lab} top-features:")
                for feat, imp in top_feats:
                    print(f"       {feat:<25s} {imp:6.3f}")
