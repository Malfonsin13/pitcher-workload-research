#!/usr/bin/env python3
"""
LIDOM / Winter Leagues pitch-level scraper + RE288 + pitch RV (Savant-style)
+ extra kinematic fields for trajectory/Stuff work.

Fixes:
- StatsAPI /v1/schedule REQUIRES sportId. Winter Leagues sportId=17.
- Chunk schedule queries to avoid missing games in long windows.
- Force PitchType to string to prevent merge dtype errors.

Adds:
- release_pos_y = 60.5 - extension
- release_pos_x/z from coordinates x0/z0
- vx0/vy0/vz0 and ax/ay/az from pitchData.coordinates
- VAA computed at plate using kinematics defined at y=50ft
- AxisDiff: angle difference between observed movement axis (pfxX,pfxZ) and spinDirection
- ArmAngle: geometric baseline using pitcher height
- Expected_IVB/HB: league-average induced vert / horiz break by PitchType

IMPORTANT CHANGE (this request):
- PitchResult is now the GRANULAR StatsAPI pitch call (e.g. "Called Strike", "Swinging Strike", etc.)
- PitchResultBucket keeps your previous rolled-up buckets (CSW, Ball, BIP-Out, etc.)
- Added swing/contact flags: Swung, Whiff, CalledStrike, FoulBall, FoulTip, InPlay, CSW
"""

def season_window(season: int) -> Tuple[date, date]:
    """
    Default window: Sep 1 of SEASON through today.
    (Sep 1 buffer helps if games start late Sep; you can override via START_DATE/END_DATE env vars.)
    """
    start = date(season, 9, 1)
    end = date.today()
    return start, end

_default_start, _default_end = season_window(SEASON)

START_DATE = date.fromisoformat(os.environ["START_DATE"]) if "START_DATE" in os.environ else _default_start
END_DATE   = date.fromisoformat(os.environ["END_DATE"])   if "END_DATE"   in os.environ else _default_end

LEAGUE_ID  = int(os.getenv("LEAGUE_ID", "131"))

# ------------------------- Helpers -------------------------

# Some feeds provide isBall/isStrike/isInPlay booleans; we still keep string fallbacks.
BALL_DESCS = {
    "Ball",
    "Ball In Dirt",
    "Intent Ball",
    "Automatic Ball",
    "Pitchout",
    "Blocked Ball",
}

STRIKE_DESCS = {
    "Called Strike",
    "Swinging Strike",
    "Swinging Strike (Blocked)",
    "Foul Tip",
    "Foul",
    "Foul Bunt",
    "Foul Pitchout",
    "Missed Bunt",
    "Automatic Strike",
}

INPLAY_DESCS = {
    "In play, out(s)",
    "In play, no out",
    "In play, run(s)",
}

def parse_height_to_inches(h: Any) -> Optional[float]:
    """Parse StatsAPI height strings like 6' 2" into inches."""
    if h is None:
        return None
    s = str(h).strip()
    if not s:
        return None
    import re
    m = re.search(r"(\d+)\s*'\s*(\d+)", s)
    if not m:
        return None
    ft = float(m.group(1))
    inch = float(m.group(2))
    return ft * 12.0 + inch

def compute_vaa_from_y50(vy0: Any, vz0: Any, ay: Any, az: Any, y0: float = 50.0) -> Optional[float]:
    """
    Vertical Approach Angle (deg) at plate (y=0).
    Solve time from y=y0 -> y=0:
        0 = y0 + vy0*t + 0.5*ay*t^2
    then vy(t), vz(t), VAA = atan2(vz_plate, -vy_plate).
    """
    try:
        vy0 = float(vy0); vz0 = float(vz0); ay = float(ay); az = float(az)
        y0  = float(y0)
    except Exception:
        return None

    if abs(ay) < 1e-8:
        if abs(vy0) < 1e-8:
            return None
        t = -y0 / vy0
        if t <= 0:
            return None
    else:
        a = 0.5 * ay
        b = vy0
        c = y0
        disc = b*b - 4*a*c
        if disc < 0:
            return None
        sdisc = math.sqrt(disc)
        t1 = (-b - sdisc) / (2*a)
        t2 = (-b + sdisc) / (2*a)
        ts = [t for t in (t1, t2) if t > 0]
        if not ts:
            return None
        t = min(ts)

    if not (0.05 <= t <= 1.2):
        return None

    vy_plate = vy0 + ay*t
    vz_plate = vz0 + az*t
    return float(np.degrees(np.arctan2(vz_plate, -vy_plate)))

def axis_diff_deg(spin_dir: Any, pfx_x: Any, pfx_z: Any) -> Optional[float]:
    """Angular diff (0–180) between observed movement axis and spinDirection."""
    try:
        sd = float(spin_dir)
        mx = float(pfx_x)
        mz = float(pfx_z)
    except Exception:
        return None
    obs = (np.degrees(np.arctan2(mx, mz)) + 360.0) % 360.0
    diff = abs(obs - (sd % 360.0)) % 360.0
    return float(min(diff, 360.0 - diff))

def geometric_arm_angle(p_throws: str, release_pos_x: Any, release_pos_z: Any, height_in: float = 74.0) -> Optional[float]:
    """Baseline arm-angle estimate using shoulder point at 70% of height. Negative for LHP."""
    try:
        height_in = float(height_in)
        shoulder_z = 0.70 * height_in / 12.0  # feet
        dx = float(release_pos_x)
        dz = float(release_pos_z) - shoulder_z
        if abs(dx) < 0.15:
            return None
        ang = float(np.degrees(np.arctan2(dz, abs(dx))))
        return -ang if (p_throws or "").upper() == "L" else ang
    except Exception:
        return None

def cnt_str(b: int, s: int) -> str:
    return f"{b}-{s}"

def parse_pitch_outcome(details: dict) -> Dict[str, Any]:
    """
    Returns granular PitchResult + derived flags + rolled-up bucket.
    StatsAPI pitch event 'details' often contains:
      - details.description and/or details.call.description
      - isBall/isStrike/isInPlay booleans
    """
    # most common human string
    desc = details.get("description")
    if not desc:
        desc = (details.get("call") or {}).get("description")

    # boolean hints (when present)
    has_is_ball   = "isBall" in details
    has_is_strike = "isStrike" in details
    has_is_inplay = "isInPlay" in details

    is_ball  = bool(details.get("isBall"))   if has_is_ball   else (desc in BALL_DESCS)
    is_strike= bool(details.get("isStrike")) if has_is_strike else (desc in STRIKE_DESCS or desc in INPLAY_DESCS)
    in_play  = bool(details.get("isInPlay")) if has_is_inplay else (desc in INPLAY_DESCS or (isinstance(desc, str) and desc.startswith("In play")))

    foul_tip = (desc == "Foul Tip")
    foul_ball = (isinstance(desc, str) and "Foul" in desc and not foul_tip and not in_play)

    whiff = desc in ("Swinging Strike", "Swinging Strike (Blocked)", "Missed Bunt")
    called_strike = desc in ("Called Strike", "Automatic Strike")

    # swung definition: anything where batter offered / made contact OR ball put in play
    swung = in_play or whiff or foul_tip or foul_ball or (isinstance(desc, str) and desc.startswith("In play"))

    # CSW definition used by many models: called strikes + swinging strikes + foul tips
    csw = called_strike or whiff or foul_tip

    # bucket (your dashboard-friendly rollups)
    if desc in ("Hit By Pitch",):
        bucket = "HBP"
    elif in_play:
        if desc == "In play, out(s)":
            bucket = "BIP-Out"
        elif desc == "In play, run(s)":
            bucket = "BIP-Hit"
        else:
            bucket = "BIP-Reached"
    elif foul_ball:
        bucket = "Foul"
    elif is_ball:
        bucket = "Ball"
    elif csw:
        bucket = "CSW"
    elif is_strike:
        bucket = "Strike"
    else:
        bucket = desc

    return {
        "PitchResult": desc,
        "PitchResultBucket": bucket,
        "Swung": swung,
        "Whiff": whiff,
        "CalledStrike": called_strike,
        "FoulTip": foul_tip,
        "FoulBall": foul_ball,
        "InPlay": in_play,
        "CSW": csw,
        "IsBall": is_ball,
        "IsStrike": is_strike,
    }


def runners_after_from_play(play: dict) -> tuple[dict, int, int]:
    """
    Returns: (bases_after_play, runs_scored_on_play, outs_added_on_play)
    Robust to LIDOM quirks where batter is sometimes missing from runners.
    """
    runners = play.get("runners") or []
    bases_after = {1: False, 2: False, 3: False}
    runs = 0
    outs_added = 0

    for r in runners:
        mv = r.get("movement") or {}
        det = r.get("details") or {}

        end = mv.get("end")
        if end in ("1B", "2B", "3B"):
            bases_after[int(end[0])] = True

        # scoring
        if end == "score" or det.get("isScoringEvent") is True:
            runs += 1

        # outs
        if det.get("isOut") is True or mv.get("isOut") is True or end == "out":
            outs_added += 1

    # Prefer explicit outs count if present
    res = play.get("result") or {}
    if res.get("outs") is not None:
        try:
            outs_added = max(outs_added, int(res.get("outs", 0)))
        except Exception:
            pass
    elif res.get("isOut") is True:
        outs_added = max(outs_added, 1)

    return bases_after, runs, outs_added


# ------------------------- StatsAPI calls -------------------------

PEOPLE_CACHE: Dict[int, Dict[str, Any]] = {}

def fetch_live(game_pk: int) -> dict:
    url = f"https://statsapi.mlb.com/api/v1.1/game/{game_pk}/feed/live"
    r = sess.get(url, timeout=45)
    r.raise_for_status()
    return r.json()

def batch_people(ids: List[int]) -> Dict[int, Dict[str, Any]]:
    """Pull fullName + height for many ids, with a simple cache."""
    want = [int(i) for i in set(ids) if int(i) not in PEOPLE_CACHE]
    if not want:
        return {i: PEOPLE_CACHE[i] for i in set(ids) if i in PEOPLE_CACHE}

    url = "https://statsapi.mlb.com/api/v1/people"
    params = {"personIds": ",".join(str(i) for i in sorted(want))}
    r = sess.get(url, params=params, timeout=45)
    if not r.ok:
        return {i: PEOPLE_CACHE[i] for i in set(ids) if i in PEOPLE_CACHE}

    for p in r.json().get("people", []):
        pid = int(p.get("id"))
        PEOPLE_CACHE[pid] = {"fullName": p.get("fullName"), "height": p.get("height")}

    return {i: PEOPLE_CACHE[i] for i in set(ids) if i in PEOPLE_CACHE}

def _daterange_chunks(start: date, end: date, chunk_days: int = 25) -> List[Tuple[str, str]]:
    """Return inclusive date chunks [s,e] in ISO strings."""
    out = []
    cur = start
    while cur <= end:
        nxt = min(end, cur + timedelta(days=chunk_days))
        out.append((cur.isoformat(), nxt.isoformat()))
        cur = nxt + timedelta(days=1)
    return out

def schedule_game_pks(start: date, end: date, game_types: Optional[str] = None) -> List[int]:
    """Schedule endpoint requires sportId. Chunk the range to avoid truncation/missed dates."""
    url = "https://statsapi.mlb.com/api/v1/schedule"
    pks: List[int] = []

    for s, e in _daterange_chunks(start, end, chunk_days=25):
        params: Dict[str, Any] = {
            "sportId": SPORT_ID,
            "leagueId": LEAGUE_ID,
            "startDate": s,
            "endDate": e,
        }
        if game_types:
            params["gameTypes"] = game_types

        r = sess.get(url, params=params, timeout=45)
        if not r.ok:
            try:
                msg = r.text[:500]
            except Exception:
                msg = "<no body>"
            raise requests.HTTPError(f"{r.status_code} schedule error for {s}..{e}: {msg}", response=r)

        data = r.json()
        for d in data.get("dates", []):
            for g in d.get("games", []):
                pk = g.get("gamePk")
                if pk:
                    pks.append(int(pk))

    return sorted(set(pks))


# ------------------------- Extraction -------------------------

def harvest_pitches(game_pk: int) -> List[Dict[str, Any]]:
    live = fetch_live(game_pk)

    game_date = ((live.get("gameData") or {}).get("datetime") or {}).get("originalDate")
    game_type = ((live.get("gameData") or {}).get("game") or {}).get("type")

    box = ((live.get("liveData") or {}).get("boxscore") or {}).get("teams") or {}
    home_team = ((box.get("home") or {}).get("team") or {})
    away_team = ((box.get("away") or {}).get("team") or {})
    home_id = home_team.get("id")
    away_id = away_team.get("id")

    pitcher_team: Dict[int, int] = {}
    batter_team: Dict[int, int] = {}

    for side, tid in (("home", home_id), ("away", away_id)):
        players = ((box.get(side) or {}).get("players") or {})
        for _, rec in players.items():
            pid = (rec.get("person") or {}).get("id")
            pos = (rec.get("position") or {}).get("code")
            if pid is None:
                continue
            if pos == "1":
                pitcher_team[int(pid)] = int(tid) if tid else None
            else:
                batter_team[int(pid)] = int(tid) if tid else None

    def team_name(tid: Any) -> Optional[str]:
        if tid == home_id:
            return home_team.get("name")
        if tid == away_id:
            return away_team.get("name")
        return None

    plays = (((live.get("liveData") or {}).get("plays") or {}).get("allPlays") or [])

    rows: List[Dict[str, Any]] = []
    last_halfkey = None  # (inning, halfInning) tracker to reset bases/outs cleanly
    bases_on = {1: False, 2: False, 3: False}
    outs_in_inning = 0
    pitch_in_game = 0

    for play in plays:
        about = play.get("about") or {}
        matchup = play.get("matchup") or {}

        inning = about.get("inning")
        half = about.get("halfInning")
        at_bat_number = about.get("atBatIndex")
        start_t = about.get("startTime")

        # ---- half-inning boundary reset (prevents outs/bases from getting "stuck") ----
        halfkey = (inning, half)
        if halfkey != last_halfkey:
            bases_on = {1: False, 2: False, 3: False}
            outs_in_inning = 0
            last_halfkey = halfkey


        pitcher_id = (matchup.get("pitcher") or {}).get("id")
        batter_id  = (matchup.get("batter") or {}).get("id")
        p_throws   = ((matchup.get("pitchHand") or {}).get("code") or "")
        batter_side = ((matchup.get("batSide") or {}).get("code") or "")

        events = play.get("playEvents") or []
        pitch_events = [ev for ev in events if ev.get("type") == "pitch"]
        if not pitch_events:
            outs_in_inning = about.get("outs", outs_in_inning)
            continue

        last_idx = pitch_events[-1].get("index")

        b1_pre, b2_pre, b3_pre = int(bases_on[1]), int(bases_on[2]), int(bases_on[3])

        # Prefer feed-provided startOuts (more reliable than our tracker in these feeds)
        o_pre = about.get("startOuts", outs_in_inning)
        o_pre = int(o_pre) if o_pre is not None else int(outs_in_inning)


        # Use runner movements for bases/runs (works fine), but compute outs from 'about' when possible
        bases_after_play, runs_on_play, outs_on_play_fallback = runners_after_from_play(play)

        # ---- outs recorded on the play ----
        # These feeds are much more consistent in play.about than play.result.outs.
        end_outs = about.get("endOuts", None)
        if end_outs is None:
            # Many feeds store end-of-play outs in about["outs"]
            end_outs = about.get("outs", None)

        outs_after_play = None
        if end_outs is not None:
            try:
                outs_after_play = int(end_outs)
            except Exception:
                outs_after_play = None

        if outs_after_play is None:
            # fallback: use our runner-based out detection
            outs_after_play = int(o_pre + int(outs_on_play_fallback or 0))

        # derive outs-on-play as delta (clamped)
        outs_on_play = max(0, outs_after_play - o_pre)
        outs_after_play = min(3, max(0, outs_after_play))

        # If inning ended, bases empty
        if outs_after_play >= 3:
            bases_after_play = {1: False, 2: False, 3: False}


        # Track PRE-pitch count ourselves by shifting from the feed's POST-pitch count
        prev_balls = 0
        prev_strikes = 0

        for ev in pitch_events:
            pitch_in_game += 1

            ev_idx = ev.get("index")
            is_terminal = (ev_idx == last_idx)

            det = ev.get("details") or {}
            pdat = ev.get("pitchData") or {}
            coords = (pdat.get("coordinates") or {})
            breaks = (pdat.get("breaks") or {})
            hit = ev.get("hitData") or {}

            tinfo = det.get("type") or {}
            pitch_type = tinfo.get("description") or tinfo.get("code")

            outcome = parse_pitch_outcome(det)
            pitch_result = outcome["PitchResult"]                 # granular
            pitch_bucket = outcome["PitchResultBucket"]           # rolled-up

            # The feed's ev["count"] is POST-pitch; we want PRE-pitch for RE288.
            # So: pre = previous post (starts 0-0), post = ev["count"].
            c_post = ev.get("count") or {}
            b_post = int(c_post.get("balls", prev_balls))
            s_post = int(c_post.get("strikes", prev_strikes))

            b_pre_pitch = int(prev_balls)
            s_pre_pitch = int(prev_strikes)
            count_pre = cnt_str(b_pre_pitch, s_pre_pitch)

            # For RV/RE transitions: after-count is the post-pitch count for non-terminal pitches.
            next_count = cnt_str(b_post, s_post)

            # advance for next pitch in this AB
            prev_balls, prev_strikes = b_post, s_post

            if is_terminal:
                b1_after, b2_after, b3_after = int(bases_after_play[1]), int(bases_after_play[2]), int(bases_after_play[3])
                outs_after = int(outs_after_play)
                runs_scored = int(runs_on_play)
                next_balls_strikes = "0-0"
                play_result = (play.get("result") or {}).get("event")
            else:
                b1_after, b2_after, b3_after = b1_pre, b2_pre, b3_pre
                outs_after = o_pre
                runs_scored = 0
                next_balls_strikes = next_count
                play_result = None

            # --------- extra fields ----------
            extension = pdat.get("extension")
            release_pos_x = coords.get("x0")
            release_pos_z = coords.get("z0")

            y0_kin = coords.get("y0")
            release_pos_y = None
            if extension is not None:
                try:
                    release_pos_y = 60.5 - float(extension)
                except Exception:
                    release_pos_y = None
            if release_pos_y is None:
                release_pos_y = y0_kin

            vx0 = coords.get("vX0")
            vy0 = coords.get("vY0")
            vz0 = coords.get("vZ0")
            ax  = coords.get("aX")
            ay  = coords.get("aY")
            az  = coords.get("aZ")

            pfx_x = coords.get("pfxX")
            pfx_z = coords.get("pfxZ")

            plate_x = coords.get("pX")
            plate_z = coords.get("pZ")
            zone = pdat.get("zone")

            spin_rate = breaks.get("spinRate")
            spin_dir  = breaks.get("spinDirection")

            # NOTE: we'll revisit these keys later if they're missing for LIDOM
            ivb_in = breaks.get("breakVerticalInduced")
            hb_in  = breaks.get("breakHorizontal")

            vaa = compute_vaa_from_y50(vy0, vz0, ay, az, y0=(y0_kin if y0_kin is not None else 50.0))
            ad  = axis_diff_deg(spin_dir, pfx_x, pfx_z)

            rows.append(dict(
                game_pk=game_pk,
                GameType=game_type,
                GameDate=game_date,
                DateTime=start_t,
                inning=inning,
                halfInning=half,
                AT_BAT_NUMBER=at_bat_number,
                PITCH_NUMBER_AB=ev.get("pitchNumber"),
                PITCH_NUMBER_GAME=pitch_in_game,

                PitcherID=pitcher_id,
                BatterID=batter_id,
                p_throws=p_throws,
                BatterHand=batter_side or "All",

                PitcherTeamID=pitcher_team.get(int(pitcher_id)) if pitcher_id else None,
                BatterTeamID=batter_team.get(int(batter_id)) if batter_id else None,
                PitcherTeam=team_name(pitcher_team.get(int(pitcher_id))) if pitcher_id else None,
                BatterTeam=team_name(batter_team.get(int(batter_id))) if batter_id else None,

                PitchType=pitch_type,

                # IMPORTANT: granular vs rolled-up
                PitchResult=pitch_result,
                PitchResultBucket=pitch_bucket,

                # swing/contact flags
                Swung=outcome["Swung"],
                Whiff=outcome["Whiff"],
                CalledStrike=outcome["CalledStrike"],
                FoulTip=outcome["FoulTip"],
                FoulBall=outcome["FoulBall"],
                InPlay=outcome["InPlay"],
                CSW=outcome["CSW"],

                PLAY_RESULT=play_result,

                balls=b_pre_pitch,
                strikes=s_pre_pitch,
                outs=o_pre,
                Count=count_pre,
                next_balls_strikes=next_balls_strikes,
                is_terminal=is_terminal,

                on_1b=b1_pre, on_2b=b2_pre, on_3b=b3_pre,
                on_1b_after=b1_after, on_2b_after=b2_after, on_3b_after=b3_after,
                outs_after=outs_after,
                runs_scored_on_pitch=runs_scored,

                Velo=pdat.get("startSpeed"),
                EXTENSION=extension,
                REL_SIDE=release_pos_x,
                REL_HEIGHT=release_pos_z,

                release_pos_x=release_pos_x,
                release_pos_y=release_pos_y,
                release_pos_z=release_pos_z,

                vx0=vx0, vy0=vy0, vz0=vz0,
                ax=ax, ay=ay, az=az,

                VAA=vaa,
                AxisDiff=ad,
                ArmAngle=None,          # fill after pitcher height lookup

                PLATE_X=plate_x,
                PLATE_Z=plate_z,
                ZONE=zone,
                PFX_X=pfx_x,
                PFX_Z=pfx_z,
                SPIN_RATE=spin_rate,
                SPIN_DIR=spin_dir,
                INDUCED_VERT_BREAK=ivb_in,
                HORZ_BREAK=hb_in,

                EXIT_VELO=hit.get("launchSpeed"),
                LAUNCH_ANGLE=hit.get("launchAngle"),
            ))

        outs_in_inning = int(outs_after_play)
        if outs_in_inning >= 3:
            outs_in_inning = 0
            bases_on = {1: False, 2: False, 3: False}
        else:
            bases_on = dict(bases_after_play)

    if not rows:
        return []

    df = pd.DataFrame(rows)

    if "PitchType" in df.columns:
        df["PitchType"] = df["PitchType"].astype("string")

    people_ids = pd.unique(pd.concat([
        df["PitcherID"].dropna().astype(int),
        df["BatterID"].dropna().astype(int),
    ], ignore_index=True)).tolist()

    ppl = batch_people(people_ids)

    df["Pitcher"] = df["PitcherID"].map(lambda x: (ppl.get(int(x)) or {}).get("fullName") if pd.notna(x) else None)
    df["Batter"]  = df["BatterID"].map(lambda x: (ppl.get(int(x)) or {}).get("fullName") if pd.notna(x) else None)
    df["PitcherHeightStr"] = df["PitcherID"].map(lambda x: (ppl.get(int(x)) or {}).get("height") if pd.notna(x) else None)
    df["PitcherHeightIn"]  = df["PitcherHeightStr"].map(parse_height_to_inches)

    def _arm(r):
        h = r["PitcherHeightIn"]
        if pd.isna(h) or h is None:
            h = 74.0
        return geometric_arm_angle(r["p_throws"], r["release_pos_x"], r["release_pos_z"], height_in=float(h))

    df["ArmAngle"] = df.apply(_arm, axis=1)

    return df.to_dict("records")


# ------------------------- RE288 + RV -------------------------

def make_re288_table(pitches: pd.DataFrame) -> pd.DataFrame:
    """RE is mean runs remaining in the half-inning from the pitch's PRE state."""
    d = pitches.copy()
    d["HalfKey"] = list(zip(d["game_pk"], d["GameDate"], d["inning"], d["halfInning"]))
    d = d.sort_values(["game_pk", "inning", "halfInning", "PITCH_NUMBER_GAME"]).reset_index(drop=True)
    d["runs"] = d["runs_scored_on_pitch"].fillna(0).astype(int)

    tot = d.groupby("HalfKey")["runs"].sum().rename("inning_runs_total")
    d = d.join(tot, on="HalfKey")

    d["runs_through"] = d.groupby("HalfKey")["runs"].cumsum()
    d["runs_before"] = d["runs_through"] - d["runs"]
    d["runs_remaining_from_state"] = d["inning_runs_total"] - d["runs_before"]

    grp = (
        d.groupby(["on_1b","on_2b","on_3b","outs","Count"])["runs_remaining_from_state"]
         .agg(RE="mean", N="size")
         .reset_index()
         .rename(columns={"on_1b":"b1","on_2b":"b2","on_3b":"b3","outs":"o"})
    )

    counts = [f"{b}-{s}" for b in range(4) for s in range(3)]
    grid = pd.DataFrame([{"b1":b1,"b2":b2,"b3":b3,"o":o,"Count":c}
                         for b1 in (0,1) for b2 in (0,1) for b3 in (0,1)
                         for o in (0,1,2)
                         for c in counts])

    global_mean = float(d["runs_remaining_from_state"].mean())
    re = grid.merge(grp, on=["b1","b2","b3","o","Count"], how="left")
    re["RE"] = re["RE"].fillna(global_mean)
    re["N"] = re["N"].fillna(0).astype(int)
    return re

def make_re288_wide_table(
    re_table_long: pd.DataFrame,
    decimals: int = 2,
    add_separators: bool = True,
) -> pd.DataFrame:
    """
    Convert long RE288 table (b1,b2,b3,o,Count,RE) into the wide Excel-style matrix:
      - Rows: 0outXXX ... 2outOOO (with optional blank separator rows)
      - Columns: ordered to match screenshot
      - Values: RE rounded
    """
    d = re_table_long.copy()

    # Defensive typing
    for c in ["b1", "b2", "b3", "o"]:
        d[c] = pd.to_numeric(d[c], errors="coerce").fillna(0).astype(int)

    # Build base occupancy pattern like XXX / OXX / ...
    d["BaseState"] = (
        d["b1"].map(lambda v: "X" if v == 1 else "O")
        + d["b2"].map(lambda v: "X" if v == 1 else "O")
        + d["b3"].map(lambda v: "X" if v == 1 else "O")
    )

    # Row label like 0outXXX
    d["State"] = d["o"].astype(str) + "out" + d["BaseState"]

    # Pivot to wide
    wide = d.pivot(index="State", columns="Count", values="RE")

    # Column order from your screenshot
    count_order = ["0-2", "1-2", "0-1", "2-2", "1-1", "0-0", "1-0", "2-1", "3-2", "2-0", "3-1", "3-0"]
    wide = wide.reindex(columns=count_order)

    # Row order from your screenshot
    base_order = ["XXX", "OXX", "XOX", "OOX", "XXO", "OXO", "XOO", "OOO"]
    outs_order = [0, 1, 2]

    ordered_rows: list[str] = []
    for i, o in enumerate(outs_order):
        ordered_rows.extend([f"{o}out{bs}" for bs in base_order])
        if add_separators and i < len(outs_order) - 1:
            # separator row (unique-ish label so pandas doesn't collapse duplicates)
            ordered_rows.append(" " * (i + 1))

    wide = wide.reindex(ordered_rows)

    # Match the top-left header cell you showed
    wide.index.name = "Pitch count ->"

    # Round for display
    wide = wide.round(decimals)

    return wide


def compute_rv_re288(pitches: pd.DataFrame, re_table: pd.DataFrame) -> pd.DataFrame:
    """
    RV = runs_scored_on_pitch + RE(after) - RE(before)
    Avoid duplicate Count column by overwriting Count in temp 'after' df.
    """
    d = pitches.copy()

    re = re_table.copy()
    re["Count"] = re["Count"].astype("string")

    before = d.rename(columns={"on_1b":"b1","on_2b":"b2","on_3b":"b3","outs":"o"}).copy()
    before["Count"] = before["Count"].astype("string")
    before = before.merge(
        re[["b1","b2","b3","o","Count","RE"]],
        on=["b1","b2","b3","o","Count"],
        how="left",
        validate="m:1"
    )
    d["RE_before"] = before["RE"].astype(float)

    after = d.rename(columns={"on_1b_after":"b1","on_2b_after":"b2","on_3b_after":"b3","outs_after":"o"}).copy()
    after["Count"] = after["next_balls_strikes"].astype("string")
    after = after.merge(
        re[["b1","b2","b3","o","Count","RE"]],
        on=["b1","b2","b3","o","Count"],
        how="left",
        validate="m:1"
    )
    d["RE_after"] = after["RE"].astype(float)

    d["RV_RE288"] = d["runs_scored_on_pitch"].fillna(0).astype(float) + d["RE_after"] - d["RE_before"]
    return d


# ------------------------- Expected movement + sanity checks -------------------------


def add_expected_movement(
    pitches: pd.DataFrame,
    arm_bin_halfwidth_deg: float = 3.0,
    min_n: int = 10,
) -> pd.DataFrame:
    out = pitches.copy()
    out = out.drop(columns=["Expected_IVB", "Expected_HB", "Expected_N"], errors="ignore")

    out["PitchType"] = out["PitchType"].astype("string")
    out["p_throws"] = out["p_throws"].astype("string")
    out["ArmAngle"] = pd.to_numeric(out.get("ArmAngle"), errors="coerce")

    out["_ivb_for_exp"] = pd.to_numeric(out.get("INDUCED_VERT_BREAK"), errors="coerce")
    out["_hb_for_exp"]  = pd.to_numeric(out.get("HORZ_BREAK"), errors="coerce")


    step = 2.0 * float(arm_bin_halfwidth_deg)
    out["_ArmBin"] = np.round(out["ArmAngle"] / step) * step

    base = out.dropna(subset=["PitchType", "p_throws", "_ArmBin", "_ivb_for_exp", "_hb_for_exp"])
    g1 = (
        base.groupby(["PitchType", "p_throws", "_ArmBin"])
        .agg(_exp_ivb1=("_ivb_for_exp", "mean"), _exp_hb1=("_hb_for_exp", "mean"), _n1=("_ivb_for_exp", "size"))
        .reset_index()
    )

    base2 = out.dropna(subset=["PitchType", "p_throws", "_ivb_for_exp", "_hb_for_exp"])
    g2 = (
        base2.groupby(["PitchType", "p_throws"])
        .agg(_exp_ivb2=("_ivb_for_exp", "mean"), _exp_hb2=("_hb_for_exp", "mean"), _n2=("_ivb_for_exp", "size"))
        .reset_index()
    )

    base3 = out.dropna(subset=["PitchType", "_ivb_for_exp", "_hb_for_exp"])
    g3 = (
        base3.groupby(["PitchType"])
        .agg(_exp_ivb3=("_ivb_for_exp", "mean"), _exp_hb3=("_hb_for_exp", "mean"), _n3=("_ivb_for_exp", "size"))
        .reset_index()
    )

    out = out.merge(g1, on=["PitchType", "p_throws", "_ArmBin"], how="left")
    out = out.merge(g2, on=["PitchType", "p_throws"], how="left")
    out = out.merge(g3, on=["PitchType"], how="left")

    n1 = pd.to_numeric(out["_n1"], errors="coerce")
    n2 = pd.to_numeric(out["_n2"], errors="coerce")
    n3 = pd.to_numeric(out["_n3"], errors="coerce")

    use1 = n1 >= min_n
    use2 = (~use1) & (n2 >= min_n)
    use3 = (~use1) & (~use2) & (n3 >= min_n)

    out["Expected_IVB"] = np.where(use1, out["_exp_ivb1"], np.where(use2, out["_exp_ivb2"], out["_exp_ivb3"]))
    out["Expected_HB"] = np.where(use1, out["_exp_hb1"], np.where(use2, out["_exp_hb2"], out["_exp_hb3"]))

    expected_n = np.where(use1, out["_n1"], np.where(use2, out["_n2"], out["_n3"]))
    out["Expected_N"] = pd.to_numeric(pd.Series(expected_n, index=out.index), errors="coerce").astype("Int64")

    out = out.drop(
        columns=[
            "_ivb_for_exp",
            "_hb_for_exp",
            "_ArmBin",
            "_exp_ivb1",
            "_exp_hb1",
            "_n1",
            "_exp_ivb2",
            "_exp_hb2",
            "_n2",
            "_exp_ivb3",
            "_exp_hb3",
            "_n3",
        ],
        errors="ignore",
    )
    return out

def sanity_check_distribution(df: pd.DataFrame) -> None:
    print("\n=== Sanity check summary ===")
    print(f"Pitches: {len(df):,}  Games: {df['game_pk'].nunique():,}")

    gd = pd.to_datetime(df["GameDate"], errors="coerce")
    if gd.notna().any():
        print(f"GameDate range: {gd.min().date()} → {gd.max().date()}")
        by_day = df.assign(_gd=gd.dt.date).groupby("_gd").size()
        print(f"Days scraped: {len(by_day):,} | median pitches/day={int(by_day.median())} | max pitches/day={int(by_day.max())}")

    if "GameType" in df.columns:
        print("GameType counts:", df["GameType"].value_counts(dropna=False).to_dict())

    # outcome quick check
    if "PitchResult" in df.columns:
        top = df["PitchResult"].value_counts(dropna=False).head(15).to_dict()
        print("Top PitchResult (granular):", top)
    if "PitchResultBucket" in df.columns:
        topb = df["PitchResultBucket"].value_counts(dropna=False).to_dict()
        print("PitchResultBucket:", topb)

    cols = ["release_pos_x","release_pos_y","release_pos_z",
            "vx0","vy0","vz0","ax","ay","az",
            "VAA","AxisDiff","ArmAngle","Expected_IVB","Expected_HB"]
    keep = [c for c in cols if c in df.columns]
    for c in keep:
        s = pd.to_numeric(df[c], errors="coerce")
        if s.notna().sum() == 0:
            print(f"{c:>14}: all missing")
            continue
        q = s.quantile([0.01,0.05,0.5,0.95,0.99]).to_dict()
        miss = 100*(1 - s.notna().mean())
        print(f"{c:>14}: miss={miss:5.2f}%  p05={q[0.05]:8.3f}  p50={q[0.5]:8.3f}  p95={q[0.95]:8.3f}  min={s.min():8.3f}  max={s.max():8.3f}")


# ------------------------- Main -------------------------

def main() -> int:
    start = START_DATE
    end   = END_DATE

    pks = schedule_game_pks(start, end, game_types=(GAME_TYPES or None))
    print(f"Schedule returned {len(pks)} games from {start.isoformat()} → {end.isoformat()} "
          f"(sportId={SPORT_ID}, leagueId={LEAGUE_ID}, GAME_TYPES={GAME_TYPES or 'ALL'}).")

    if not pks:
        print("No games found. If this is wrong, try setting GAME_TYPES='R,P' or widening START_DATE.")
        return 1

    all_rows: List[Dict[str, Any]] = []
    errors: List[Tuple[int, str]] = []

    for pk in tqdm(pks, desc="Scraping games"):
        try:
            all_rows.extend(harvest_pitches(int(pk)))
        except Exception as e:
            errors.append((int(pk), str(e)))
            print(f"! game {pk} error: {e}")

    if not all_rows:
        print("No pitch rows extracted.")
        return 1

    pitches = pd.DataFrame(all_rows)

    if "PitchType" in pitches.columns:
        pitches["PitchType"] = pitches["PitchType"].astype("string")

    pitches = add_expected_movement(pitches)

    re_table_long = make_re288_table(pitches)
    re_table_wide = make_re288_wide_table(re_table_long, decimals=2, add_separators=True)

    pitches = compute_rv_re288(pitches, re_table_long)

    pitches.to_csv(OUT_PITCHES, index=False)

    # Write the wide, screenshot-style RE table
    re_table_wide.to_csv(OUT_RE, index=True, float_format="%.2f")

    # (Optional) also write the long version for debugging / merges
    # re_table_long.to_csv("lidom_re_table_long.csv", index=False)

    sanity_check_distribution(pitches)

    if errors:
        print("\n=== Games with errors (first 25) ===")
        for pk, msg in errors[:25]:
            print(f"- {pk}: {msg}")
        print(f"Total error games: {len(errors)}")

    print(f"\nWrote {OUT_PITCHES}\nWrote {OUT_RE}")
    return 0

if __name__ == "__main__":
    raise SystemExit(main())
